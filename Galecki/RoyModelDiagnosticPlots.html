for(i in 1:nrow(dat)){
  LOOmodel = lme(BP ~ method-1, 
                 data = dat[-i,],
                 random = list(subject=pdSymm(~ method-1)), 
                 weights=varIdent(form=~1|method),
                 correlation = corSymm(form=~1 | subject/repl), 
                 method="ML")

  all.res1[i]=fixef(LOOmodel)[1]
  all.res2[i]=fixef(LOOmodel)[2]

}
JS.roy1.dfbeta <- cbind(all.res1,all.res2)
